OPTION_Top Paying Jobs
What are the top-paying jobs for my role?

SELECT TOP 10 [job_id]
      ,[job_title]
      ,[job_location]
      ,[job_schedule_type]
      ,[job_posted_date]
      ,[salary_year_avg]
  FROM [sql_course].[dbo].[job_postings_fact]
  WHERE [job_title]='Data Analyst' AND 
  [job_location]='Anywhere' AND
  [salary_year_avg] IS NOT NULL
  ORDER BY [salary_year_avg] DESC

OPTION_In-Demand Skills
What are the most in-demand skills for my role?

SELECT TOP 5 [skills], COUNT(skills) AS skills_count
FROM [sql_course].[dbo].[job_postings_fact]
INNER JOIN [sql_course].[dbo].[skills_job_dim] ON [sql_course].[dbo].[job_postings_fact].job_id=[sql_course].[dbo].[skills_job_dim].job_id
INNER JOIN [sql_course].[dbo].[skills_dim] ON [sql_course].[dbo].[skills_job_dim].[skill_id]=[sql_course].[dbo].[skills_dim].skill_id
WHERE [job_title]='Data Analyst'
GROUP BY (skills)
ORDER BY [skills_count] DESC
